cmake_minimum_required(VERSION 3.12)
project(MOC CXX)


add_executable(moc mnist_moc.cpp)

find_package(Eigen3 REQUIRED)
target_link_libraries(moc PRIVATE Eigen3::Eigen)


set(CMAKE_CXX_STANDARD 17)

find_package(OpenMP REQUIRED)
if(OpenMP_CXX_FOUND)
     target_link_libraries(moc PRIVATE OpenMP::OpenMP_CXX)
endif()


target_include_directories(moc PRIVATE ${CMAKE_SOURCE_DIR}/rapidcsv)
