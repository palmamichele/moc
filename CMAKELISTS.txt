cmake_minimum_required(VERSION 3.12)

project(pybind_vid)

add_subdirectory(pybind11)
pybind11_add_module(exactdmoc exact_moc.cpp)


set(EIGEN3_INCLUDE_DIR "${CMAKE_SOURCE_DIR}/include/eigen")
target_include_directories(exactdmoc PRIVATE ${EIGEN3_INCLUDE_DIR})


set(CMAKE_CXX_STANDARD 17)

find_package(OpenMP REQUIRED)
if(OpenMP_CXX_FOUND)
     target_link_libraries(exactdmoc PRIVATE OpenMP::OpenMP_CXX)
endif()
